A simple Vector implementation



#include <array>
#include <cstddef>
template <typename T, std::size_t dimension>
using Vector = std::array<T, dimension>;
int main() {
  Vector<int, 12> s{1, 2, 3, 4, 5, 6, 7, 8, 9, 0, 1};
  return s[0];
}
